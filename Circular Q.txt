#include<stdio.h>
#define MAX 5
int queue[MAX];
int front = -1, rear = -1;
void Enqueue(int x)
{
    if ( front == (rear + 1)%MAX)
    {
        printf("Queue full");
    }
    else if( rear == -1)
    {
        front = rear = 0;
        queue[rear] = x;
    }
    else
    {
        rear =  (rear + 1)%MAX;
        queue[rear] = x;
    }
}
int Dequeue()
{
    int temp = -1;
    if ( front == -1)
        printf("Queue empty");
    else if(front == rear)
    {
        temp = queue[front];
        front = rear = -1;
    }
    else
    {
        temp = queue[front];
        front = (front + 1)%MAX;
    }
    return temp;
}
int main()
{
    return 0;
}
