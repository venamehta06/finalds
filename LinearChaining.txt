#include <stdio.h>
#include <stdlib.h>
#define SIZE 10

struct node {
    int data;
    struct node *next;
};

struct node *table[SIZE];

int hash(int key) {
    return key % SIZE;
}

void insert(int key) {
    int i = hash(key);
    struct node *new = malloc(sizeof(struct node));
    new->data = key;
    new->next = table[i];
    table[i] = new;
}

void display() {
    for (int i = 0; i < SIZE; i++) {
        printf("%d -> ", i);
        struct node *temp = table[i];
        while (temp) {
            printf("%d ", temp->data);
            temp = temp->next;
        }
        printf("\n");
    }
}

int main() {
    insert(23);
    insert(43);
    insert(13);
    insert(27);

    display();
    return 0;
}
