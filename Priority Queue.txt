#include<stdio.h>
#define MAX 5
int pq[MAX];
int n = 0;
void insert(int x)
{
    int i = 0;
    for(i = n - 1; i >=0 && pq[i] > x; i--)
        pq[i+1] = pq[i];
    pq[i+1] = x;
    n++;
}
int del()
{
    return pq[--n];
}
int main()
{
    insert(10);
    insert(2);
    insert(3);
    printf("%d", del());
}
