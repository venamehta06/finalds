#include <stdio.h>

int main() {
    int a[10], n, i, key;
    int l = 0, h, m;

    scanf("%d", &n);

    for(i = 0; i < n; i++)
        scanf("%d", &a[i]);

    scanf("%d", &key);

    h = n - 1;

    while(l <= h) {
        m = (l + h) / 2;

        if(a[m] == key) {
            printf("Found");
            return 0;
        }
        else if(a[m] < key)
            l = m + 1;
        else
            h = m - 1;
    }

    printf("Not Found");
    return 0;
}
