#include<stdio.h>
#define MAX 5
int queue[MAX];
int front = -1, rear = -1;
void Enqueue(int x)
{
    if ( rear == MAX-1)
    {
        printf("Queue full");
    }
    else if( rear == -1)
    {
        front = rear = 0;
        queue[rear] = x;
    }
    else
    {
        rear++;
        queue[rear] = x;
    }
}
int Dequeue()
{
    int temp = -1;
    if ( front == -1)
        printf("Queue empty");
    else if(front == rear)
    {
        temp = queue[front];
        front = rear = -1;
    }
    else
    {
        temp = queue[front++];
    }
    return temp;
}
int main()
{
    return 0;
}
