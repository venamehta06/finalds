#include<stdio.h>
#include<stdlib.h>
#define MAX 5
int stk[MAX];
int top = -1;
void push(int x)
{
    if ( top == MAX-1)
        printf("Stack full");
    else
    {
        top++;
        stk[top] = x;
    }
}
int pop()
{
    int temp = 0;
    if (top == -1)
        return -1;
    else
    {
        temp = stk[top];
        top--;
        return temp;
    }
}
void peek()
{
    if (top == -1)
        printf("Stack empty");
    else
        printf("%d", stk[top]);
}
void display()
{
    int i = top;
    for ( i ; i >=0; i--)
        printf("%d", stk[i]);
}
int main()
{
    return 0;
}
